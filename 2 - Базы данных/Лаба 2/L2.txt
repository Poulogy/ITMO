WITH MIN_AGE_GROUPS AS (SELECT min(AGE("Н_ЛЮДИ"."ДАТА_РОЖДЕНИЯ")) AS MINAGE
                        FROM "Н_ЛЮДИ"
                            JOIN "Н_УЧЕНИКИ" ON "Н_ЛЮДИ"."ИД" = "Н_УЧЕНИКИ"."ЧЛВК_ИД"
                        WHERE "Н_УЧЕНИКИ"."ГРУППА" = '6509')
SELECT "Н_УЧЕНИКИ"."ГРУППА", avg(AGE("Н_ЛЮДИ"."ДАТА_РОЖДЕНИЯ"))
FROM "Н_УЧЕНИКИ"
    JOIN "Н_ЛЮДИ" ON "Н_ЛЮДИ"."ИД" = "Н_УЧЕНИКИ"."ЧЛВК_ИД"
GROUP BY "Н_УЧЕНИКИ"."ГРУППА"
HAVING MIN_AGE_GROUPS.MINAGE < avg(AGE("Н_ЛЮДИ"."ДАТА_РОЖДЕНИЯ"))
